def clb(n,k):
    dp = [0]*(n+1)
    dp[0] = 1 # 保证当站在第0层的时候也算一种情况，不然无法推着走
    for i in range(n+1):

3、最长上升子序列 (LIS)核心思路和前面最大子段和类似，都是记录每个数字所在位置对应数组前面部分包含自身在内最大序列长，这样后面的位置某个数要求以它为最大值的前面最大序列长 (小) 就等于他前面所有小于它本身所有数字所在位置记录的序列最长记录的一个最大值，然后还是使用一个变量 mx_su 动态更新维护每个位置数字记录的它前面部分小于等于它本身的最长序列记录值的最大值。

def mx_sul(nms):
    dp = [1]*len(nms) # 每个位置记录的自身前面部分小于等于自己的序列包含自身
    mx_su = 0
    for i in range(len(nms)):
        for j in range(i): # 0到i-1，当前自身位置已经记录
            if nms[i] > nms[j]: # 只取小于自己的数所记录的序列长，不然也不是递增
                dp[i] = max(dp[i],dp[j]+1) # 加1，是加上自身
        mx_su = max(dp[i],mx_su)
    return mx_su
nms = list(map(int,input().split()))
print(mx_sul(nms))
关键还是需要找到用什么状态量化题意 (直接对每个位置存储题目要求结果的值，还是可以记录别的最后简单处理)，然后是用这种量化状态的方式能不能用程序语句表示对应状态转移方式。
拓展：上述解法时间复杂度是 O (n**2), 如果数据规模变大，可能超时，实际解题如果只求 (最长序列值)，而不要求打印所有满足这个值的具体序列，可以使用贪心 + 二分思想具体思路：使用一个数组 d 存储遍历数组过程中当前值能放到最大位置，如果取出数大于 d 数组每个值，就说明当前遍历数之前部分至少已经满足最长序列可以加 1, 则对应在数组 d 添加一个当前元素，否则找到当前遍历数在 d 数组中对应第一个大于等于遍历数的位置，并将 d 数组中这个位置的数替换为这个新的相对较小的遍历数，而替换是保证当前组尽可能的可以保证继续添加新的元素 (核心是大于最后一个元素，但是最后一个元素如果遇到相对小的就可以变得更小，从而增大增加新元素的可能，但是它又不能随意替换，而要保证它要大于 d 记录数组倒数第二个数不然这个序列就不是有序的了，因此又关涉到倒数第二个数的维护，同理倒数第二个数的更新维护与倒数第三个数有关，因此每次拿到数都要给它放到合适位置，而不仅仅是与最后一个位置的简单判断)

import bisect # 这是python标准库模块，蓝桥杯可以直接使用
def mx_sl(nms):
    d = []
    for x in nms:
        pos = bisect.bisect_left(d,x)
        if pos == len(d):
            d.append(x)
        else:
            d[pos] = x
    return len(d)
关键就是事先设定目标上升序列的记录组 d, 遍历原数组过程中，为了尽可能保证序列最长，目标记录组每个位置的数就应该尽可能小 (虽然影响序列长的是添加元素步骤，只看最后一个数，但是前面数可能也可以有更好的选择，而使得后面出现完全不一样的序列，比如这组数 34123 它突然来一个 1，并且后面三个数明显可以组成更好的情况)，所以需要把遍历新得到的数放到合适位置，从而保证当前结果序列最后一个数可以调换的下限更小，从而达到尽可能增长 d 数组长度的目的，开始还以为他这个结果得到的 d 也是原数组对应顺序挑取的一个序列，实际并不是，比如 2 3 1 4 5，最后得到的实际是 1 3 4 5 而不是实际的 2 3 4 5，它这个 1 换到这里也好，换到第三个位置也好 (如果可以的话) 只是为了一步步服务于最后一个数 (当前序列最大数) 尽可能缩小，如果这里换成一个小的数比如 3，那么最后一个 5 可选的调换数就可以是 4 了，这样再来一个 5 就直接追加到 d 数组尾部，数组长度加一，从而最长上升序列长度加一 (因为实际只关心最终 d 数组长度和是否大于 d 最后一个数以及 d 最后一个数的更新维护)抽象至极尼玛
至于如何还原这个上升子序列，自己研究半天也没搞明白，看了标准的解法，可以使用回溯记录前驱，代码如下。关键就是定义一个 ix 数组，专门动态记录当前 d 对应每个数在原数组位置，然后针对原数组每一个数，设置一个 pre 用于标记它的前一个数，默认 - 1，用于回溯，首先容易知道最终得到的 d 一定是上升序列最后一个数，因为对于这个 d 实际上它的每一个元素就是以当前元素下标 + 1 为长度的原数组中的上升子序列中整体最小 (末位数最小) 的那个序列对应的末尾数，所以 d 最后一个数一定就是该最长序列 (整体最小) 的末尾数，可以直接作为回溯的起点。最后根据 pre 数组，使用 cr 遍历回溯即可。 至于这个 pre 到底记录的是啥，因为这个 d 是不断变化的，对于 d 中某一个元素 (除开第一个), 如果我们在遍历得到它的时候，因为我们只关心前面的部分，所以可以不看原数组后面部分，那么当前这个 pos 它就是只关心部分的最后一个 pos, 更换 d 中对应元素，分为两种情况，如果这个 pos 不是 d 末尾元素时，它虽然换了某个数，但是不影响原来 d 中后面元素已经记录的前驱，只需要取出此时 d 最后一个元素作为起点，就可以正常还原只关心部分最长上升序列，当前如果前面那个元素不是最终序列元素，那么该元素必然没有坚持到更新最后一个元素，自然过滤，所以对于最终正确的最长序列对应路径此时前面那个数必然是最终序列一部分，记录其为当前前驱即可；如果是末尾元素，那么此时这个元素就是当前只关心部分最长子序列末尾元素，对应 d 前一个元素自然是最长序列之一，正常标记即可。

import bisect
def lis_seq(nms):
    d,ix = [],[]
    pr = [-1]*len(nms)
    for i,x in enumerate(nms):
        ps = bisect.bisect_left(d,x) # 找第一个大于等于x的
        if ps == len(d):
            d.append(x)
            ix.append(i)
        else:
            d[ps] = x
            ix[ps] = i
        if ps>0:
            pr[i] = ix[ps-1]
    rs = []
    cr = ix[-1]
    while cr != -1:
        rs.append(nms[cr])
        cr = pr[cr]
    return rs[::-1]
nms = list(map(int,input().split()))
print(*lis_seq(nms))

4、01 背包 (一维优化)概念：对于物件列表每个物件，不能多次选，换句话说就是选与不选的问题。思路：当前背包容量最大价值 V，对于当前物件的选取，记录每一个加上该物件容量 w [i] 后背包当前容量 d [i] 对应价值，维护选取该物件后容量的对应价值，此时容量 i 最少为 w [i], 最多为容量 V，从后往前逆序遍历，对每一个容量 i，对应价值更新为不选该物件之前背包容量对应价值 (d [v-w [i]]) 加上选的该物件价值 v [i] 的和与自身已存价值 d [i] 的一个比较维护。最后每一个 d 记录的每一个值，就是对应能选取的最大价值情况。

def zer_pck(n,V,w,v): # n：可选物品件数，V：背包总容量，w：物品重量列表，v:物品价值列表
	dp = [0]*(V+1) # 注意这里是V+1,以存储最大容量V对应价值
	for i in range(n):
		cw = w[i]
		cv = v[i]
		for j in range(V,cw-1,-1): # 从后往前推
			dp[j] = max(dp[j],dp[j-cw]+cv)
	return dp[V]
关键就是对于每一轮选取，前面的 dp 记录必然已经是那个容量在已经完成轮次选取的对应最大价值，感觉相当于倒着装，不正向思考剩的还能装多少对应价值变为多少，而是倒着看装了某物件后这个容量对应价值是否变大（所以必须逆序遍历，如果顺序遍历就是前面已经选了该物件，然后后面又用那个选了的最大价值更新，就错了，因为只能选一次）还是挺神奇的
5、 完全背包 (一维)概念：相对于 01 背包，每一个物件可以多次选思路：主要还是前面状态结果推后面状态结果，对于每一件物品，每一个当前可能容量状态对应价值等于不选该物件前容量状态记录价值加上当前物件价值，但是不同于 01 背包的是，在遍历这每一个容量状态时采用顺序遍历的方式，这里有两种理解方式，从整体状态看如果可以任意选取当前物件，对于前面的每一个容量状态都去比较一下看哈装了这个物件价值是否变大 (和 01 背包一样，这里不是拿自身当前容量加，而是减去这个物品容量后的前面已记录此容量最大价值情况来加物件价值进行比较)，会变大就更新，这样就顺理成章的多次选取；其次，从前面所述 01 背包反面看，从后往前遍历保证用的是不选该物件前容量状态对应价值，对每一个状态更新从计算顺序上保证了选取的一次性，那么反过来顺着遍历，就是在计算时间上的自然延后，用于当前容量状态维护的不选该物件时对应容量对应价值就已经是完成选取与否判断后结果 (即允许多次选取)，且最后 dp [V] 就是对应最大值

import sys
import math
def standard_complete_pack(V, items):
    """标准完全背包一维DP（正序）"""
    dp = [0] * (V + 1)
    for w, v in items:
        for j in range(w, V + 1):
            dp[j] = max(dp[j], dp[j - w] + v)
    return dp[V]

def user_method(V, items):
    """
    尽可能复现你代码中的逻辑：
    1. 找出性价比最高的物品 mx_cho
    2. 计算所有物品重量乘积 pdt，如果 pdt <= V 则进行分解
    3. 计算所有物品重量的最大公因数 dw
    4. 缩放容量和重量后做完全背包
    5. 返回结果
    """
    # 找出性价比最高的物品（用交叉乘法避免浮点误差）
    mx_cho = items[0]
    max_ratio_num = mx_cho[1]  # 分子
    max_ratio_den = mx_cho[0]  # 分母
    for w, v in items[1:]:
        # 比较 v/w 与当前最大，即 v * max_ratio_den > max_ratio_num * w
        if v * max_ratio_den > max_ratio_num * w:
            mx_cho = (w, v)
            max_ratio_num = v
            max_ratio_den = w
        elif v * max_ratio_den == max_ratio_num * w and w < mx_cho[0]:
            # 性价比相同，取重量小的
            mx_cho = (w, v)
    # 计算所有物品重量乘积，若超过V则pdt=0
    pdt = 1
    for w, _ in items:
        pdt *= w
        if pdt > V:
            pdt = 0
            break
    # 计算最大公因数
    dw = items[0][0]
    for w, _ in items[1:]:
        dw = math.gcd(dw, w)
    if dw == 0:
        dw = 1
    bas = 0
    if pdt != 0:
        lv = V - pdt
        # 用性价比最高的物品尽量填满 lv
        bas = (lv // mx_cho[0]) * mx_cho[1]
        # 剩余部分模 mx_cho[0] 再加回乘积，作为新的V
        V = (lv % mx_cho[0]) + pdt
    # 缩放后容量
    mdl = V // dw
    d = [0] * (mdl + 1)
    scaled_items = [(w // dw, v) for w, v in items]
    for cw, cv in scaled_items:
        for j in range(cw, mdl + 1):
            d[j] = max(d[j], d[j - cw] + cv)
    return d[mdl] + bas

    d = [0]*(M+1)
    d[0] = 1
    for cp in cps:
        for i in range(cp,M+1):

V = int(input())
n = int(input())
def pck_lf():
    dp = [0]*(V+1)
    for i in range(n):
        v = int(input())
        for j in range(V,v-1,-1):
            dp[j] = max(dp[j],dp[j-v]+v)
    return V-dp[V]

def dit_rcd():
    bt = 1 # 重量为0，可以不选，对应一种选法
    msk = (1<<(V+1))-1 # 截取码，截取超过V位情况，用于且操作，超过V位都是0,则且操作时自然过滤
    for _ in range(n):
        i = int(input())
        bt = bt|(bt<<i) # 或操作，保证记录选取新物件(左移对应位)和选之前两种状态(都可以选到所以必须记录为1)

    return bt.bit_length()-1 # 使用bit_lenth方法找到该二进制最高为1的位(所以前面必须过滤超出的无效位)

V = int(input())
n = int(input())
vs = [0]*n
suf = [0]*(n+1) # 后缀和开n+1空间,suf[n]=0,表示当前及当前之后元素和
for i in range(n):
    vs[i] = int(input())
vs.sort()
for i in range(n-1,-1,-1):  # 提前存储后缀和
    suf[i] = suf[i+1]+vs[i]
rs = 0
def bac_src(si,cr):
    nonlocal rs
    if cr+suf[si]<=V:
        rs = max(rs,cr+suf[si])
        return
    if cr>rs:
        rs = cr
    pr = -1
    for i in range(si,n):
        if vs[i] == pr: # 上一个回退元素正是这个直接跳过，因为这里只看重量
            continue
        if cr+vs[i]>V:
            return
        bac_src(i+1,cr+vs[i])
        pr = vs[i]
bac_src(0,0)
print(V-rs)

3、 排数问题 (保留特定子序列) p12377递归形式的状态传递思路：对每一个数位从高位起使用循环递归遍历，对每一层递归，在循环外，先定义 ct，初始化为 0，累加上当前层当前位置各个数字，及确定当前位数后，之后所有数字位对应情况的不完全数的情况和 (递归调用下一个位置对应的不同状态状态)，循环结束后返回这个 ct, 给上一层的 ct, 从而达到计数的效果 (这种结构还是比较神奇，循环外计数设置为 0，循环累加传入不同值对应下一层递归，最后返回这个计数给上一层累加)。然后这里递归，涉及四个参数，及一个状态对应四个标量，stu: 到当前数位已经记录的匹配状态长 (及该位置前序列对应目标序列的最大匹配长，在递归开始判断是否等于目标序列长，等于就说明该数字不管后面位数字如何都满足匹配，非法，直接返回 0，从而省掉大批数字的遍历),ps: 当前层对应数位 (等于最大位数长时返回 1，表示一种情况)，lgu: 布尔值，表示前面数位与范围下限数的匹配 (如果当前 ps 位层传过来的是 True，表示当前位置前面所有数与范围下限数完全对应，为了保证当前遍历数在范围内，对当前位数字就不能随便选，需要根据这个状态位确定可选范围)，hgu：布尔值，类似于 lgu, 表示与上限数的匹配 (如果为真，则当前位不能大于上限数当前位对应数)。在递归内部前面完成简单判断剪枝后，就根据前面的 lgu，hgu 状态，确定当前位置数的上下限，用于控制后面循环次数，都为假时为 0-9, 否则分别根据情况取上下限范围数对应数位值。最后在循环内部开头更新递归的四个新状态量，对于 stu，单独定义一个状态转换函数 (传入当前对应状态，当前遍历选取数)。根据当前状态量记录，和当前选的数，进行选择判断，如果状态为 2，表示当前截取序列已经最大匹配到目标列的第 2 个数，则如果当前选取数等于目标列第三个数，则状态更新返回 3，否则保持返回 2; 如果传入状态为 4，则直接返回 4，其他状态数更新类似。对于 lgu, 如果 lgu 为真且当前位等于前面的算出的数位下限，则保持这种完全比对 (True)，否则为假，hgu 更新类似，用于下一层数位遍历范围计算。

from functools import lru_cache
def nx_staus(st,cd):
    if st==0:
        return 1 if cd ==2 else 0
    elif st==1:
        return 2 if cd == 0 else 1
    elif st==2:
        return 3 if cd == 3 else 2
    elif st==3:
        return 4 if cd == 3 else 3
    else:
        return 4
@lru_cache(maxsize = None)
def slv():
    lf,rt = input().split()
    lgt = len(lf)
    def dfs(stu,ps,lsu,hsu):
        if stu==4:
            return 0
        if ps==lgt:
            return 1
        lo = int(lf[ps]) if lsu else 0
        hi = int(rt[ps]) if hsu else 9
        tol = 0
        for cd in range(lo,hi+1):
            nx_stu = nx_staus(stu,cd)
            nx_lsu = lsu and (cd==lo)
            nx_hsu = hsu and (cd==hi)
            tol += dfs(nx_stu,ps+1,nx_lsu,nx_hsu)
        return tol
    ans = dfs(0,0,True,True)
    return ans
print(slv())
